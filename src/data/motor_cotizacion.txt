# Documentación de las fórmulas principales

FORMULAS
# Prima Base Vida Término = Cobertura * Tasa_Edad
# Prima Base Vida Completa = Cobertura * Tasa_Edad * 2.5  
# Prima Base Vida Ahorro = Cobertura * Tasa_Edad * 3.0

# Prima Final = Prima_Ajustada * (1 + Recargos) * (1 - Descuentos)
# Cobertura Recomendada = Ingresos_Mensuales * 12 * Años_Ingresos_Recomendados
